func MergeSort(arr []int) []int {
    // –ë–∞–∑–æ–≤—ã–π —Å–ª—É—á–∞–π: –º–∞—Å—Å–∏–≤—ã –∏–∑ 0 –∏–ª–∏ 1 —ç–ª–µ–º–µ–Ω—Ç–∞ —É–∂–µ –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω—ã

        return arr
    }
    
    // –î–ï–õ–ï–ù–ò–ï (log n —Ä–∞–∑)
    // –î–µ–ª–∏–º –º–∞—Å—Å–∏–≤ –ø–æ–ø–æ–ª–∞–º - —ç—Ç–æ –ø—Ä–æ–∏—Å—Ö–æ–¥–∏—Ç –ø—Ä–∏–º–µ—Ä–Ω–æ log‚ÇÇn —Ä–∞–∑

    left := MergeSort(arr[:mid])   // O(n log n) - —Ä–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤
    right := MergeSort(arr[mid:])  // O(n log n) - —Ä–µ–∫—É—Ä—Å–∏–≤–Ω—ã–π –≤—ã–∑–æ–≤
    
    // –°–õ–ò–Ø–ù–ò–ï (O(n) –Ω–∞ –∫–∞–∂–¥–æ–º —É—Ä–æ–≤–Ω–µ —Ä–µ–∫—É—Ä—Å–∏–∏)
    return merge(left, right)      // O(n)
}

// merge - —Ñ—É–Ω–∫—Ü–∏—è —Å–ª–∏—è–Ω–∏—è –¥–≤—É—Ö –æ—Ç—Å–æ—Ä—Ç–∏—Ä–æ–≤–∞–Ω–Ω—ã—Ö –º–∞—Å—Å–∏–≤–æ–≤
func merge(left, right []int) []int {
    result := make([]int, 0, len(left)+len(right))
    i, j := 0, 0
    
    // –°—Ä–∞–≤–Ω–∏–≤–∞–µ–º —ç–ª–µ–º–µ–Ω—Ç—ã –∏–∑ –¥–≤—É—Ö –º–∞—Å—Å–∏–≤–æ–≤ –∏ –¥–æ–±–∞–≤–ª—è–µ–º –º–µ–Ω—å—à–∏–π
    for i < len(left) && j < len(right) {
        if left[i] <= right[j] {
            result = append(result, left[i])
            i++
        } else {
            result = append(result, right[j])
            j++
        }
    }
    
    // –î–æ–±–∞–≤–ª—è–µ–º –æ—Å—Ç–∞–≤—à–∏–µ—Å—è —ç–ª–µ–º–µ–Ω—Ç—ã (–µ—Å–ª–∏ –µ—Å—Ç—å)
    result = append(result, left[i:]...)
    result = append(result, right[j:]...)
    
    return result
}

–ü—Ä–∏–º–µ—Ä 2: QuickSort (—Ç–∞–∫–∂–µ O(n log n) –≤ —Å—Ä–µ–¥–Ω–µ–º —Å–ª—É—á–∞–µ)
```go
package main

import (
    "fmt"
    "math/rand"
)

// QuickSort - –±—ã—Å—Ç—Ä–∞—è —Å–æ—Ä—Ç–∏—Ä–æ–≤–∫–∞
func QuickSort(arr []int) []int {
    if len(arr) < 2 {
        return arr
    }
    
    left, right := 0, len(arr)-1
    
    // –í—ã–±–∏—Ä–∞–µ–º –æ–ø–æ—Ä–Ω—ã–π —ç–ª–µ–º–µ–Ω—Ç (pivot)
    pivot := rand.Intn(len(arr))
    
    // –ü–µ—Ä–µ–º–µ—â–∞–µ–º pivot –≤ –∫–æ–Ω–µ—Ü
    arr[pivot], arr[right] = arr[right], arr[pivot]
    
    // –†–∞–∑–¥–µ–ª–µ–Ω–∏–µ: —ç–ª–µ–º–µ–Ω—Ç—ã –º–µ–Ω—å—à–µ pivot —Å–ª–µ–≤–∞, –±–æ–ª—å—à–µ - —Å–ø—Ä–∞–≤–∞
    for i := range arr {
        if arr[i] < arr[right] {
            arr[left], arr[i] = arr[i], arr[left]
            left++
        }
    }
    
    // –í–æ–∑–≤—Ä–∞—â–∞–µ–º pivot –Ω–∞ –ø—Ä–∞–≤–∏–ª—å–Ω—É—é –ø–æ–∑–∏—Ü–∏—é
    arr[left], arr[right] = arr[right], arr[left]
    
    // –†–µ–∫—É—Ä—Å–∏–≤–Ω–æ —Å–æ—Ä—Ç–∏—Ä—É–µ–º –ª–µ–≤—É—é –∏ –ø—Ä–∞–≤—É—é —á–∞—Å—Ç–∏
    QuickSort(arr[:left])
    QuickSort(arr[left+1:])
    
    return arr
}

func main() {
    arr := []int{10, 80, 30, 90, 40, 50, 70}
    fmt.Println("–î–æ QuickSort:", arr)
    
    QuickSort(arr)
    fmt.Println("–ü–æ—Å–ª–µ QuickSort:", arr)
}
```
